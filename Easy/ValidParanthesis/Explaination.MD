We are given a string `s` containing only the characters:

- `(`, `)`, `{`, `}`, `[`, `]`

We need to determine if the input string is **valid**.
A string is valid if:

1. Every opening bracket has a corresponding closing bracket.
2. Brackets close in the correct order (Last Opened â†’ First Closed).

var isValid = function (s) {
  const stack = [];

  for (let i = 0; i < s.length; i++) {
    const char = s[i];

    // Case 1: Opening brackets â†’ push to stack
    if (char === "{" || char === "(" || char === "[") {

    }
    // Case 2: Closing brackets â†’ check stack
    else if (char === "}" || char === ")" || char === "]") {
      if (isEmpty(stack)) {
        return false;
      }

      const top = stack.pop();

      // If closing bracket doesn't match the top of stack â†’ invalid
      if (
        (char === ")" && top !== "(") ||
        (char === "}" && top !== "{") ||
        (char === "]" && top !== "[")
      ) {
        return false;
      }
    }
  }

  // At the end, stack must be empty (all brackets matched)
  return isEmpty(stack);
};

function isEmpty(s) {
  return s.length === 0;
}
